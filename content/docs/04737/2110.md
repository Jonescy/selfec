    ```c++
    #include <iostream>
    
    using namespace std;
    
    template<class T>
    void f(T a[], int n) {
        int i;
        for (int i = 0; i < n / 2; i++) {
            T t;
            t = a[i];
            a[i] = a[n - 1 - i];
            a[n - 1 - i] = t;
        }
    
        for (i = 0; i < n; i++)
            cout << a[i] << " ";
        cout << endl;
    }
    
    
    int main() {
        int a[6] = {1, 2, 3, 4, 5, 6}, i;
        double b[3] = {1.2, 3.4, 5.6};
        f(a,6);
        f(b,3);
        return 0;
    }
    ```
